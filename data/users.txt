public static void login(Scanner sc) {
    System.out.print("Email: ");
    String email = sc.nextLine();
    System.out.print("Contraseña: ");
    String contrasenia = sc.nextLine();

    Path usersFile = Paths.get("data", "users.txt");

    if (!Files.exists(usersFile)) {
        System.out.println("No hay usuarios registrados.");
        return;
    }

    try {
        java.util.List<String> lineas = Files.readAllLines(usersFile);
        boolean iniciar_sesion_correcto = false;

        for (String linea : lineas) {
            String[] datos = linea.split(";");
            if (datos[0].equals(email) && datos[1].equals(contrasenia)) {
                iniciar_sesion_correcto = true;
                System.out.println("Ha iniciado sesión");
                menu_notas(sc, email);
                break;
            }
        }

        if (!iniciar_sesion_correcto) {
            System.out.println("Email y/o contraseña incorrecto.");
        }
    } catch (IOException e) {
        System.out.println("Hubo un error.");
    }
}
